``` python
        try:  
            response=self.client.chat.completions.create(  
                model=self.model,  
                messages=messages,  
                temperature=temperature,  
                stream=True  
            )

            correct_content=[]  
            for chunk in response:  
                if not chunk.choices:  
                    continue  
                content=chunk.choices[0].delta.content or ""  
                print(content,end="",flush=True)  
                correct_content.append(content)  
            print()  
            return "".join(correct_content)  
        except Exception as e:  
            print(f"模型调用失败{e}")
```

## **构建通用工具执行器**

``` python
from typing import Dict, Any

class ToolExecutor:
    """
    一个工具执行器，负责管理和执行工具。
    """
    def __init__(self):
        self.tools: Dict[str, Dict[str, Any]] = {}
        """
        为什么这么写？
        因为实际所有的工具是这样的类型：
        {
    "search": {                           # 外层：工具名称（str）
        "description": "搜索工具...",      # 内层：工具信息
        "function": <function search>     # 内层：工具函数
    },
    "calculator": {                       # 外层：另一个工具名称
        "description": "计算器工具...",
        "function": <function calc>
    }
}
        """

    def registerTool(self, name: str, description: str, func: callable):
        """
        向工具箱中注册一个新工具。
        """
        if name in self.tools:
            print(f"警告:工具 '{name}' 已存在，将被覆盖。")
        self.tools[name] = {"description": description, "func": func}
        print(f"工具 '{name}' 已注册。")

    def getTool(self, name: str) -> callable:
        """
        根据名称获取一个工具的执行函数。
        """
        return self.tools.get(name, {}).get("func")

    def getAvailableTools(self) -> str:
        """
        获取所有可用工具的格式化描述字符串。
        """
        return "\n".join([
            f"- {name}: {info['description']}" 
            for name, info in self.tools.items()
        ])

	
```

## 构建ReAct类
**ReAct的核心循环**：==不断地“格式化提示词 -> 调用LLM -> 执行动作 -> 整合结果”，直到任务完成或达到最大步数限制。==

- 导入client和tool执行器
- 初始化变量：client,tool_executer,max_steps（防止智能体陷入无尽循环，消耗资源）,history
- 创建智能体入口run函数,其while循环构建了ReAct范式主体
- 创建对于prompt提到的输出内容Thought,Action提取函数
- 创建对于Action中的工具名称和工具输入的提取函数

``` python
from client import HelloAgent  
from tool_build import ToolExecutor,search  
import re  
class ReActAgent:  
    def __init__(self,llm_client=HelloAgent,tool_executor=ToolExecutor,max_steps:int=5):  
        self.llm_client=llm_client  
        self.tool_executor=tool_executor  
        self.max_steps=max_steps  
        self.history=[]  
    def run(self,question:str):  
        self.history=[]  
        current_step=0  
        while current_step<=self.max_steps:  
            current_step+=1  
            print(f"第{current_step}步")  
            tool_desc=self.tool_executor.getAvaliableTools()  
            history_str="\n".join(self.history)  
            prompt=REACT_PROMPT_TEMPLATE.format(  
                tools=tool_desc,  
                question=question,  
                history=history_str  
            )  
            messages=[{"role":"user","content":prompt}]  
            response_text=self.llm_client.think(messages)  
            if not response_text:  
                print("模型无响应")  
                break  
            thought,action=self._parse_output(response_text)  
            if thought:  
                print(f"思考：{thought}")  
            if not action:  
                print("警告：未能解析出有效Action,流程终止")  
            if action.startswith("Finish"):  
                final_answer=self._parse_action_input( action)  
                print(f"最终答案：{final_answer}")  
                return final_answer  
            tool_name,tool_input=self._parse_action(action)  
            if not tool_name or not tool_input:  
                self.history.append("Observation: 无效的Action格式，请检查。")  
                continue  
            print(f"行动：{tool_name}[{tool_input}]")  
            tool_function=self.tool_executor.getTool(tool_name)  
            if not tool_function:  
                observation=f"错误：未找到名为{tool_name}的工具"  
            else:  
                observation=tool_function(tool_input)  
            print(f"观察：{observation}")  
            self.history.append(f"Action: {action}")  
            self.history.append(f"Observation: {observation}")  
        print("已达到最大步数，流程终止")  
        return None  
  
    def _parse_output(self,text:str):  
        thought_match=re.search(r"Thought:\s*(.*?)(?=\nAction:|$)",text,re.DOTALL)  
        action_match=re.search(r"Action:\s*(.*?)$",text,re.DOTALL)  
        thought=thought_match.group(1).strip() if thought_match else None  
        action=action_match.group(1).strip() if action_match else None  
        return thought,action  
    def _parse_action(self,action_text:str):  
        match=re.match(r"(\w+)\[(.*)\]",action_text,re.DOTALL)  
        tool_name=match.group(1)  
        tool_input=match.group(2)  
        if match:  
            return tool_name,tool_input  
        return None,None  
  
    def _parse_action_input(self, action_text: str):  
        match = re.match(r"\w+\[(.*)\]", action_text, re.DOTALL)  
        return match.group(1) if match else ""  
if __name__ == '__main__':  
    client=HelloAgent()  
    tool_executor=ToolExecutor()  
    search_desc="一个基于Serpapi的实战网页搜索引擎工具。它会智能地解析搜索结果，优先返回直接答案或知识图谱信息"  
    tool_executor.register_tool("Search",search_desc,search)  
    agent=ReActAgent(client,tool_executor)  
    question="2026年苹果最新一款手机是哪一款？它的主要卖点是什么"  
    agent.run(question)

```
## 特点：
ReAct的本质就是边做边看结果，每次思考都影响工具的调用，并利用工具的调用结果影响下一步思考，再调用
- 优点：
	- 能看到Agent的决策过程，方便调试
	- 能根据observation结果调整下一步行动输入（如换搜索词）
	- 可结合工具
	*弥补了LLM本身知识不够新，计算不够准，不能访问外部信息的缺陷*
- 缺点：
	- 过于依赖模型能力，倘若模型不按照提示词要求输出，则影响正则解析结果
	- 效率低。每一步流程都需要调用一次大模型，耗时且成本高
	- 过于依赖提示词。提示词细微的变化都可能影响输出
	- 容易陷入无限循环

# 构建Plan-and-Sove(先规划后执行)

为了解决思维链在处理多步骤、复杂问题时容易“偏离轨道”的问题。
适用于那些结构性强、可以被清晰分解的复杂任务
## 两个阶段：
### 1. 规划阶段：
分解问题，指定清晰、分步骤的行动计划
### 2. 执行阶段：
严格按照行动计划，按步骤执行。

``` python
class Executor:  
    def __init__(self,llm_client):  
        self.llm_client=llm_client  
    def execute(self,question:str,plan:List[str])->str:  
        history=''  
        print("正在执行计划")  
        for i,step in enumerate(plan):  
            print(f"正在执行第{i+1}步：{step}")  
            prompt=EXECUTOR_PROMPT_TEMPLATE.format(  
                question=question,  
                plan=plan,  
                history=history,  
                current_step=step  
            )  
            messages=[{'role':'user','content':prompt}]  
            response_text=self.llm_client.think(messages) or ""  
            history+=f"第{i+1}步：{step}\n{response_text}\n"  
            print(f"第{i+1}步执行完毕")  
        final_answer=response_text  
        return final_answer
```

# Reflection
![[智能体经典范式构建-1780920915028.webp]]
## **核心工作流程**：
- 执行——反思——优化
- **执行**：使用ReAct或者Plan-and-Solve尝试完成任务，生成初步解决方案
- **反思**：调用独立且额外设置提示词的大模型充当评审员，以多维度评估第一次的执行初稿
- **优化**：智能体将*初稿*和*反馈*作为新的上下文，再次调用大模型，要求根据反馈对初稿进行修改，生成更完善的修订稿
## 优势：
- 内部纠错回路，不再依赖外部工具反馈，修正高层次逻辑策略错误
- 将一次性任务执行过程转为持续优化过程，提升成功率和答案质量
- 为智能体构建短期记忆，整个执行反思优化过程形成了大模型的经验记录，更进一步，这个记忆系统还可以是**多模态的**，允许智能体反思和修正文本以外的输出（如代码、图像等），为构建更强大的多模态智能体奠定了基础。
## 构建Reflection
为了在实战中体现 Reflection 机制，我们将引入记忆管理机制.
Reflection 的核心在于**迭代**，而**迭代的前提是能够记住**之前的尝试和获得的反馈。因此，一个“**短期记忆**”模块是实现该范式的必需品。这个记忆模块将负责存储每一次“执行-反思”循环的完整轨迹。
### 1. 构建记忆模块
- 使用一个列表 `records` 来按顺序存储每一次的行动和反思。
- `add_record` 方法负责向记忆中添加新的条目。
- `get_trajectory` 方法是核心，它将记忆轨迹“序列化”成一段文本，可以直接插入到后续的提示词中，为模型的反思和优化提供完整的上下文。
- `get_last_execution` 方便我们获取最新的“初稿”以供反思。
``` python
from typing import Optional,List,Dict,Any  
class Memory:  
    def __init__(self):  
        self.records:List[Dict[str,Any]]=[]  
  
    def add_record(self,record_type:str,content:str):  
        record={'type':f'{record_type}','content':f'{content}'}  
        self.records.append(record)  
        print(f"添加了记录：{record}")  
    def get_trajectory(self)->str:  
        trajectory_part=[]  
        for record in self.records:  
            if record['type']=='execution':  
                trajectory_part.append(f"——上一轮尝试（代码）——：\n{record['content']}")  
            elif record['type']=='reflection':  
                trajectory_part.append(f"————评审员评审：\n{record['content']}")  
        return "\n\n".join(trajectory_part)  
    def get_last_execution(self)->Optional[str]:  
        for record in reversed(self.records):  
            if record['type']=='execution':  
                return record['content']  
        return None

```

``` python
from client import HelloAgent  
from memory import Memory  
class ReflectionAgent:  
    def __init__(self,llm_client,max_iteration:int=3):  
        self.llm_client=llm_client  
        self.memory=Memory()  
        self.max_iterations=max_iteration  
    def run(self,task:str):  
        print(f"——开始处理任务——：{task}")  
        print("\n---正在进行初始尝试---")  
        initial_prompt=INITIAL_PROMPT_TEMPLATE.format(task=task)  
        initial_code=self.get_llm_response(initial_prompt)  
        self.memory.add_record("execution",initial_code)  
        for i in range(self.max_iterations):  
            print(f"————正在第{i+1}/{self.max_iterations}轮尝试————")  
            reflect_prompt=REFLECT_PROMPT_TEMPLATE.format(task=task,code=initial_code)  
            feedback=self.get_llm_response(reflect_prompt)  
            self.memory.add_record("reflection",feedback)  
            if "无需改进" in feedback:  
                print("算法已经达到最佳，无需再优化")  
                break  
            print("正在生成新代码尝试...")  
            refine_prompt=REFINE_PROMPT_TEMPLATE.format(task=task,last_code_attempt=initial_code,feedback=feedback)  
            refine_code=self.get_llm_response(refine_prompt)  
            self.memory.add_record("execution",refine_code)  
        final_code=self.memory.get_last_execution()  
        print(f"————最终代码————：\n{final_code} \n")  
        return final_code  
  
  
    def get_llm_response(self,prompt:str)->str:  
        messages=[{'role':'user','content':prompt}]  
        response_text=self.llm_client.think(messages)  
        return response_text  
if __name__ == '__main__':  
    client=HelloAgent()  
    agent=ReflectionAgent(client)  
    agent.run("编写一个Python函数，找出1到n之间所有的素数 (prime numbers)。")
```

## Reflection成本收益分析：
### 主要成本：
- **模型调用开销**：每一轮迭代，至少额外调用两次大预言模型（一次反思，一次优化），迭代多轮，导致API调用成本和计算资源成倍增加
- **任务延迟显著提高**:Reflection 是一个串行过程，每一轮的优化都必须等待上一轮的反思完成。这使得任务的总耗时显著延长，不适合对实时性要求高的场景。
- **提示工程复杂度上升**:如我们的案例所示，Reflection 的成功在很大程度上依赖于高质量、有针对性的提示词。为“执行”、“反思”、“优化”等不同阶段设计和调试有效的提示词，需要投入更多的开发精力。
### 核心收益：
- 提升方案质量
- 提高最终结果可靠性——自带内部纠错
