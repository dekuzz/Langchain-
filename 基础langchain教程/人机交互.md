# 一、代码展示
``` python
import os  
import uuid  
from dotenv import load_dotenv  
from langchain_openai import ChatOpenAI  
from langchain.agents import create_agent  
from langchain.agents.middleware import HumanInTheLoopMiddleware  
from langchain_core.tools import tool  
from langgraph.checkpoint.memory import InMemorySaver  
from langgraph.types import Command  
# 加载模型配置  
_ = load_dotenv()

# 配置大模型服务  
llm = ChatOpenAI(  
    api_key=os.getenv("DASHSCOPE_API_KEY"),  
    base_url=os.getenv("DASHSCOPE_BASE_URL"),  
    model="qwen3-coder-plus",  
)  
  
# 工具函数  
@tool  
def get_weather(city: str) -> str:  
    """Get weather for a given city."""  
    return f"It's always sunny in {city}!"  
  
@tool  
def add_numbers(a: float, b: float) -> float:  
    """Add two numbers and return the sum."""  
    return a + b  
  
@tool  
def calculate_bmi(weight_kg: float, height_m: float) -> float:  
    """Calculate BMI given weight in kg and height in meters."""  
    if height_m <= 0 or weight_kg <= 0:  
        raise ValueError("height_m and weight_kg must be greater than 0.")  
    return weight_kg / (height_m ** 2)  
  
# 创建带工具调用的Agent  
tool_agent = create_agent(  
    model=llm,  
    tools=[get_weather, add_numbers, calculate_bmi],  
    middleware=[  
        HumanInTheLoopMiddleware(   
            interrupt_on={  
                # 无需触发人工审批  
                "get_weather": False,  
                # 需要审批，且允许approve,edit,reject三种审批类型  
                "add_numbers": True,  
                # 需要审批，允许approve,reject两种审批类型  
                "calculate_bmi": {"allowed_decisions": ["approve", "reject"]},  
            },  
            description_prefix="Tool execution pending approval",  
        ),  
    ],  
    checkpointer=InMemorySaver(),  
    system_prompt="You are a helpful assistant",  
)

# 运行Agent  
config = {'configurable': {'thread_id': str(uuid.uuid4())}}  
result = tool_agent.invoke(  
    {"messages": [{  
        "role": "user",  
        "content": "我身高180cm，体重180斤，我的BMI是多少"  
        # "content": "what is the weather in sf"  
    }]},  
    config=config,  
)  
  
# result['messages'][-1].content  
result.get('__interrupt__')

# Resume with approval decision  
result = tool_agent.invoke(  
    Command(  
        resume={"decisions": [{"type": "approve"}]}  # or "edit", "reject"  
    ),   
    config=config  
)  
  
result['messages'][-1].content

```

# 二、代码学习
## 1. `interrupt_on`&`description_prefix`
### interrupt_on
- 作用：定义**哪些工具**需要拦截，以及**拦截到什么程度**（给用户多少权限）。
- 三种典型配置：
	- **`False` (静默执行)**：直接通过，人类无感知，不产生拦截。
    
	- **`True` (完全放行权限)**：触发拦截，人类可以 **[同意 / 拒绝 / 修改参数]**。
    
	- **`字典格式` (精细化授权)**：例如 `{"allowed_decisions": ["approve", "reject"]}`，限制人类只能做特定操作（如禁止修改参数）。
### description_prefix
- 作用：定义**拦截时显示的标题**。用于告诉审核员当前为何暂停，提供操作上下文。

## 2. str(uuid.uuid4())
- 作用：每次运行脚本时，自动“新建一个全新且随机的空白存档槽”
- 拆分理解：
	- uuid.uuid4()：`uuid` 是 Python 内部专门用来生成“通用唯一识别码”的工具库。其中的 `uuid4` 专门负责根据随机数来生成一串长代码。**[它的生成逻辑保证了这串字符在世界上极大概率是独一无二的。]**
	- `thread_id` 需要的是最基础的文本，所以套上 `str()` 将它转换为普通的字符串类型。
## 3. result.get('**interrupt**')
- **作用**：
	- **断点调试**：`interrupt` 来**审查大模型的中间思考结果**，然后再决定是否放行
``` python
# 运行 Agent
result = tool_agent.invoke(user_message, config=config)
interrupt_data = result.get('interrupt')

if interrupt_data:
    print("🛑 Agent 已暂停！正在审查 AI 的企图...")
    
    # 作为一个 AI 工程师，你需要提取并打印这些信息来做 Debug
    # 假设我们能从 interrupt_data 中解析出调用的工具和参数（具体取决于框架版本）
    print(f"调用的工具: {interrupt_data.description_prefix}") 
    
    # 【关键调试点】检查 AI 提取的参数准不准
    # 比如用户说“体重180斤”，AI 有没有正确换算成 kg，还是直接传了 180？
    # print(f"AI 准备传入的参数: {interrupt_data.args}")
```


-  **本地模拟**：用 `interrupt` 配合 Python 内置的 `input()` 函数，给自己做一个简易的命令行审批流。
``` python
if interrupt_data:
    # 在终端里给自己弹出一个输入框
    decision = input("请输入你的决定 (approve / reject / edit): ")
    
    if decision in ["approve", "reject"]:
        # 根据你在终端的输入，组装恢复指令
        resume_command = Command(
            resume={"decisions": [{"type": decision}]}
        )
        
        # 带着决定和记忆（config），让 AI 继续跑
        final_result = tool_agent.invoke(resume_command, config=config)
        print("最终结果:", final_result['messages'][-1].content)
	
```
