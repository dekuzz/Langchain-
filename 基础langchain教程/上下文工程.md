# 动态修改系统提示词
## 使用state 管理上下文
``` python 
import os  
import uuid  
import sqlite3  
  
from typing import Callable  
from dotenv import load_dotenv  
from dataclasses import dataclass  
from langchain_openai import ChatOpenAI  
from langchain.tools import tool, ToolRuntime  
from langchain.agents import create_agent  
from langchain.agents.middleware import dynamic_prompt, wrap_model_call, ModelRequest, ModelResponse, SummarizationMiddleware  
from langgraph.graph import StateGraph, MessagesState, START, END  
from langgraph.checkpoint.memory import InMemorySaver  
from langgraph.store.memory import InMemoryStore  
from langgraph.store.sqlite import SqliteStore  
  
# 加载模型配置  
_ = load_dotenv()  
  
# 加载模型  
llm = ChatOpenAI(  
    api_key=os.getenv("DASHSCOPE_API_KEY"),  
    base_url=os.getenv("DASHSCOPE_BASE_URL"),  
    model="qwen3-coder-plus",  
    temperature=0.7,  
)

@dynamic_prompt  
def state_aware_prompt(request: ModelRequest) -> str:  
    # request.messages is a shortcut for request.state["messages"]  
    message_count = len(request.messages)  
  
    base = "You are a helpful assistant."  
  
    if message_count > 6:  
        base += "\nThis is a long conversation - be extra concise."  
  
    # 临时打印base看效果  
    print(base)  
  
    return base  
  
agent = create_agent(  
    model=llm,  
    middleware=[state_aware_prompt]  
)  
  
result = agent.invoke(  
    {"messages": [  
        {"role": "user", "content": "广州今天的天气怎么样？"},  
        {"role": "assistant", "content": "广州天气很好"},  
        {"role": "user", "content": "吃点什么好呢"},  
        {"role": "assistant", "content": "要不要吃香茅鳗鱼煲"},  
        {"role": "user", "content": "香茅是什么"},  
        {"role": "assistant", "content": "香茅又名柠檬草，常见于泰式冬阴功汤、越南烤肉"},  
        {"role": "user", "content": "auv 那还等什么，咱吃去吧"},  
    ]},  
)  
  
for message in result['messages']:  
    message.pretty_print()

```
### 使用场景：
- **生命周期：** 绑定于单一线程（Thread）或单次工作流。
- **核心驱动逻辑：** “当前对话的上下文进展到了什么阶段？”
- 动态修改提示词的典型场景：
	- **上下文衰减控制 (Context Window Management)：** 当对话轮数或消耗的 Token 逼近模型注意力阈值时，向系统提示词强制注入约束指令（如：“对话已过长，强制要求极度精简输出”），以防止模型产生幻觉或遗忘初始指令。
	    
	- **多步任务路由追踪 (Multi-step Routing)：** 在复杂的状态机（StateGraph）中，根据当前节点的状态变量变化，动态切换模型的“人设”或工作焦点。例如，在收集信息阶段设定为“提问者”，在信息齐备进入下一节点时，将提示词切换为“总结者”。
	    
	- **会话内行为纠偏 (In-session Course Correction)：** 监控当前 State 中的错误重试次数。如果模型连续调用工具失败或遭到用户连续否定，动态注入“禁止尝试猜测，立即请求人类干预”的熔断指令。
## 使用store管理上下文
``` python
@dataclass  
class Context:  
    user_id: str  
  
@dynamic_prompt  
def store_aware_prompt(request: ModelRequest) -> str:  
    user_id = request.runtime.context.user_id  
  
    # Read from Store: get user preferences  
    store = request.runtime.store  
    user_prefs = store.get(("preferences",), user_id)  
  
    base = "You are a helpful assistant."  
  
    if user_prefs:  
        style = user_prefs.value.get("communication_style", "balanced")  
        base += f"\nUser prefers {style} responses."  
    return base  
  
store = InMemoryStore()  
  
agent = create_agent(  
    model=llm,  
    middleware=[store_aware_prompt],  
    context_schema=Context,  
    store=store,  
)  
  
# 预置两条偏好信息  
store.put(("preferences",), "user_1", {"communication_style": "Chinese"})  
store.put(("preferences",), "user_2", {"communication_style": "Korean"})

# 用户1喜欢中文回复  
result = agent.invoke(  
    {"messages": [  
        {"role": "system", "content": "You are a helpful assistant. Please be extra concise."},  
        {"role": "user", "content": 'What is a "hold short line"?'}  
    ]},  
    context=Context(user_id="user_1"),  
)  
  
for message in result['messages']:  
    message.pretty_print()
    
# 用户2喜欢韩文回复  
result = agent.invoke(  
    {"messages": [  
        {"role": "system", "content": "You are a helpful assistant. Please be extra concise."},  
        {"role": "user", "content": 'What is a "hold short line"?'}  
    ]},  
    context=Context(user_id="user_2"),  
)  
  
for message in result['messages']:  
    message.pretty_print()
```
### 使用场景：
**生命周期：** 跨线程、跨会话、永久持久化（需执行显式覆写或删除操作）。
**核心驱动逻辑：** “关于当前交互实体（用户/组织），存在哪些全局不变的既定事实？”
**动态修改提示词的典型场景：**

- **全局个性化对齐 (Personalization & Alignment)：** 绕过前端显式输入，在底层静默读取用户的静态画像（偏好的语言、专业背景、输出格式要求），并将其作为基准指令注入提示词。这消除了每次新开会话都需要重复“设定角色”的冗余操作。
    
- **核心资产与事实留存 (Fact Retention)：** 提取过去历史中沉淀的高价值事实（如用户正在推进的特定项目架构）。当用户在新会话中提出宽泛问题时，提示词提供背景铺垫，约束模型在特定的事实框架内作答。
    
- **租户/项目级规则隔离 (Multi-tenant Rule Isolation)：** 在多租户架构下，根据 Store 里的项目 ID，动态加载该项目专属的业务术语表（Glossary）或合规红线，确保不同租户的数据与规则绝对隔离。
## 使用Runtime管理上下文
``` python 
@dataclass  
class Context:  
    user_role: str  
    deployment_env: str  
  
@dynamic_prompt  
def context_aware_prompt(request: ModelRequest) -> str:  
    # Read from Runtime Context: user role and environment  
    user_role = request.runtime.context.user_role  
    env = request.runtime.context.deployment_env  
  
    base = "You are a helpful assistant."  
  
    if user_role == "admin":  
        base += "\nYou can use the get_weather tool."  
    else:  
        base += "\nYou are prohibited from using the get_weather tool."  
  
    if env == "production":  
        base += "\nBe extra careful with any data modifications."  
  
    return base  
  
@tool  
def get_weather(city: str) -> str:  
    """Get weather for a given city."""  
    return f"It's always sunny in {city}!"  
  
agent = create_agent(  
    model=llm,  
    tools=[get_weather],  
    middleware=[context_aware_prompt],  
    context_schema=Context,  
    checkpointer=InMemorySaver(),  
)
# 利用 Runtime 中的两个变量，动态控制 System prompt# 将 user_role 设为 admin，允许使用天气查询工具  
config = {'configurable': {'thread_id': str(uuid.uuid4())}}  
result = agent.invoke(  
    {"messages": [{"role": "user", "content": "广州今天的天气怎么样？"}]},  
    context=Context(user_role="admin", deployment_env="production"),  
    config=config,  
)  
  
for message in result['messages']:  
    message.pretty_print()
    
# 若将 user_role 改为 viewer，则无法使用天气查询工具  
config = {'configurable': {'thread_id': str(uuid.uuid4())}}  
result = agent.invoke(  
    {"messages": [{"role": "user", "content": "广州今天的天气怎么样？"}]},  
    context=Context(user_role="viewer", deployment_env="production"),  
    config=config,  
)  
  
for message in result['messages']:  
    message.pretty_print()
result['messages']
```
### 使用场景：
**生命周期：** 瞬态（Transient），仅存活于当前正在执行的这一次请求（Invoke/Run）周期内。 **核心驱动逻辑：** “当前执行瞬间的外部环境和物理边界条件是什么？” **动态修改提示词的典型场景：**

- **动态权限与能力阻断 (RBAC & Capability Boundaries)：** 校验发起请求的用户身份令牌（Token）。对于普通用户，在提示词中硬编码“禁止使用数据写入工具”；对于管理员，则解除限制。这是在 LLM 层进行安全防御的关键手段。
    
- **时空与环境基准校准 (Environmental Grounding)：** 注入不适合硬编码的高频变化变量。例如，系统的实时时间戳（用于让模型准确解析“昨天”、“明天”），或当前的部署环境标志（如：“当前为生产环境，严禁输出测试用的模拟数据”）。
    
- **临时外挂载荷 (Ephemeral Payload)：** 用户在单次请求中临时附带的图片、上传的文件或当前前端页面的 DOM 树。这些数据体积大且无长期存储价值，仅在本次计算时通过 Runtime 解析并转化为文本，作为临时参考资料拼接进提示词。
